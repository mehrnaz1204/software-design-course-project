@startuml

' موجودیت‌های خارجی (سمت چپ)
class Driver {
  + requestEntry()
  + requestExit()
}

class Camera {
  + readPlate()
}

class EntryOperator {
  + confirmNormalEntry()
}

class ExitOperator {
  + confirmEmergencyExit()
}

class FinancialManager {
  + requestFinancialReport()
}

class TechnicalManager {
  + requestTechnicalReport()
}

' موجودیت‌های خارجی (سمت راست)
class PoliceSystem {
  + queryPlateStatus()
}

class SmsSystem {
  + sendPaymentConfirmation()
}

class BankSystem {
  + processPayment()
}

' مخازن داده (D1 و D2)
class AuthorizedCarsDB {
  - authorizedPlates: List
  + checkAuthorization(plate)
  + updateAuthorization(plate)
}

class TrafficRecordsDB {
  - records: List
  + saveEntryRecord(plate, time)
  + saveExitRecord(plate, time)
  + getTrafficCount(plate)
}

' فرایندهای سطح یک (۱.۰ تا ۶.۰)
class PlateVerification {
  + verifyPlate(plate)
}

class PlateStatusInquiry {
  + inquiryFromPolice(plate)
  + checkWithDB(plate)
}

class EntryManagement {
  + allowEntry(plate)
  + registerEntryTime(plate)
}

class TrafficRegistration {
  + recordTraffic(plate, type)
}

class PaymentManagement {
  + calculateFee(duration)
  + processPayment(amount)
}

class EmergencyExitManagement {
  + handleEmergencyExit(plate)
}

' ========== ارتباطات (دقیقاً بر اساس DFD) ==========

' ارتباط با فرایند ۱.۰
Driver --> PlateVerification : sends plate
Camera --> PlateVerification : sends plate

' ارتباط بین فرایند ۱.۰ و ۲.۰
PlateVerification --> PlateStatusInquiry : sends plate for inquiry

' ارتباط فرایند ۲.۰ با موجودیت‌های خارجی و مخازن
PlateStatusInquiry --> PoliceSystem : queries plate
PlateStatusInquiry --> AuthorizedCarsDB : checks authorization

' ارتباط فرایند ۲.۰ با ۳.۰
PlateStatusInquiry --> EntryManagement : sends authorization result

' ارتباط فرایند ۳.۰ با موجودیت‌ها
EntryOperator --> EntryManagement : confirms entry
EntryManagement --> Driver : allows entry

' ارتباط فرایند ۳.۰ با ۴.۰
EntryManagement --> TrafficRegistration : sends entry info

' ارتباط فرایند ۴.۰ با مخزن D2
TrafficRegistration --> TrafficRecordsDB : saves record

' ارتباط فرایند ۴.۰ با ۵.۰ (برای محاسبه هزینه خروج)
TrafficRegistration --> PaymentManagement : sends duration for fee

' ارتباط فرایند ۵.۰ با موجودیت‌های خارجی و مخازن
PaymentManagement --> BankSystem : requests payment
BankSystem --> PaymentManagement : returns confirmation
PaymentManagement --> SmsSystem : sends payment SMS
PaymentManagement --> Driver : sends receipt

' ارتباط فرایند ۶.۰ با موجودیت‌ها و مخازن
ExitOperator --> EmergencyExitManagement : confirms emergency exit
EmergencyExitManagement --> TrafficRecordsDB : saves emergency exit record
EmergencyExitManagement --> Driver : notifies exit

' ارتباطات مدیریتی (گزارش)
FinancialManager --> PaymentManagement : requests financial report
TechnicalManager --> TrafficRecordsDB : requests technical report

@enduml
